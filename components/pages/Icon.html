使用icon属性来定义图标的样式。
  <ba-icon icon="check-circle"></ba-icon>
  <ba-icon icon="camera"></ba-icon>
  <ba-icon icon="github"></ba-icon>
  <ba-icon icon="headphones"></ba-icon>
  <ba-icon icon="tool"></ba-icon>
  <ba-icon icon="thumbs-up"></ba-icon>
